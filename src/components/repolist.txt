import React from 'react'

const Repos = ({ repos, loading }) => {
  if (loading) {
    return <h2>Loading...</h2>;
  }

  return (
    <ul className='List-group mb-4'>
      {repos.map(repo => (
        <li key={repo.id} className='List-group-item'>
          {repo.name}
        </li>
      ))}
    </ul>
  );
};



export default Repos;
